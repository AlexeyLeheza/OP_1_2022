#include <iostream>
#include <string>
#include <fstream>
#include <windows.h>
using namespace std;
string InputFile()//Введення початкового текстового файлу
{
    string file, str;
    cout << "Введіть ім'я файлу:";
    getline(cin, file);
    cout << "Введіть файл. Для завершення вводу натисніть Ctrl+S:" << endl;
    char ends = 19;
    ofstream InputFile(file);
    do
    {
        getline(cin, str);
        if (str.find(ends) == string::npos)//читання всього рядочка до кінця
            InputFile << str << '\n';
        else
        {
            int pos = str.find(ends);
            str.erase(pos, str.length());
            InputFile << str;
            InputFile.close();
            return file;
        }
    } while (true);
}
string Changed_file(string file)
{
    string newFile, str, first_word, sp;
    char symbol, ch;
    ifstream inFile(file);
    cout << "Введіть ім'я нового текстового файлу:";
    getline(cin, newFile);
    ofstream outFile(newFile);
    if (inFile)//Перевірка безпомилкового відкриття файлу
    {
        while (!inFile.eof())
        {
            int space = 0, i = 0;
            str = "";
            sp = "";
            getline(inFile, str);
            while (str[i] == ' ')
            {
                space += 1;
                i++;
            }
            while (isalpha(str[i]) || isdigit(str[i]))
            {
                i++;
            }
            symbol = str[i - 1];
            sp.insert(0, space, ' ');
            outFile << sp << str << ' ' << symbol << endl;
        }

    }
    else
    {
        cout << "файл не був успішно відкритий або файл пустий";
    }
    outFile.close();
    inFile.close();
    return newFile;
}
void Print_file(string file)//Функція вивиодить текстовий файл
{
    string s;
    ifstream inFile(file);
    if (inFile)
    {
        cout << "Файл:" << file << endl;

        {
            getline(inFile, s);
            cout << s << endl;
        }
    }
    else
    {
        cout << "файл не був успішно відкритий або файл пустий";

}
